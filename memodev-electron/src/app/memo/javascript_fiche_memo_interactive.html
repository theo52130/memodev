Niveau Débutant

Concepts de Base

JavaScript est un langage de programmation interprété, orienté objet et de haut niveau. Voici quelques concepts fondamentaux :

  • Client-side : Le code JavaScript s'exécute principalement dans le navigateur du client
  • Dynamiquement typé : Les types de variables peuvent changer pendant l'exécution
  • Événementiel : Basé sur des événements et des callbacks
  • Single-threaded : S'exécute sur un seul thread avec une boucle d'événements
  • Multiplateforme : Fonctionne sur différents environnements (navigateurs, Node.js)
  • ECMAScript : Basé sur la norme ECMAScript (ES6, ES7, etc.)

// Tagged templates
function highlight(strings, ...values) {
  return strings.reduce((result, str, i) => {
    return result + str + (values[i] ? `${values[i]}` : '');
  }, '');
}
const text = highlight`Bonjour ${name}, vous avez ${30} ans.`;

  return numbers.reduce((total, n) => total + n, 0);
}

// Spread operator

# Classes (ES6)
class Personne {
  constructor(nom, age) {
    this.nom = nom;
    this.age = age;
  }
  
  // Méthode d'instance
  sePresenter() {
    return `Je m'appelle ${this.nom} et j'ai ${this.age} ans`;
  }
  
  // Méthode statique
  static creerAdulte(nom) {
    return new Personne(nom, 18);
  }
  
  // Getter
  get estMajeur() {
    return this.age >= 18;
  }
}

// Héritage
class Employe extends Personne {
  constructor(nom, age, poste) {
    super(nom, age);
    this.poste = poste;
  }
  
  sePresenter() {
    return `${super.sePresenter()} et je travaille comme ${this.poste}`;
  }
}

Programmation asynchrone avancée

# Promises
// Création d'une promesse
const promise = new Promise((resolve, reject) => {
  const success = true;
  if (success) {
    resolve("Opération réussie");
  } else {
    reject("Opération échouée");
  }
});

// Utilisation de promesses
promise
  .then(result => {
    console.log(result);
    return "Nouvelle valeur";
  })
  .then(result => {
    console.log(result);
  })
  .catch(error => {
    console.error(error);
  })
  .finally(() => {
    console.log("Exécuté quoi qu'il arrive");
  });

# Promise combinators
// Promise.all - attend que toutes les promesses soient résolues
Promise.all([fetchUser(), fetchPosts(), fetchComments()])
  .then(([user, posts, comments]) => {
    // Tous les résultats sont disponibles ici
  })
  .catch(error => {
    // Si une seule promesse échoue, tout échoue
  });

// Promise.race - retourne la première promesse résolue ou rejetée
Promise.race([fetchData(), timeout(5000)])
  .then(result => {
    // La première promesse à se résoudre
  });

// Promise.allSettled (ES2020) - attend que toutes les promesses soient terminées
Promise.allSettled([fetchA(), fetchB(), fetchC()])
  .then(results => {
    // Résultats contient le statut et la valeur/raison de chaque promesse
  });

// Promise.any (ES2021) - retourne la première promesse résolue avec succès
Promise.any([fetchA(), fetchB(), fetchC()])
  .then(firstResult => {
    // Premier résultat réussi
  })
  .catch(error => {
    // Toutes les promesses ont échoué
  });

# Async/Await (ES2017)
// Fonction asynchrone
async function fetchUserData() {
  try {
    // Attendre la résolution de la promesse
    const response = await fetch('https://api.example.com/user');
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const data = await response.json();
    return data;
  } catch (error) {
    console.error('Une erreur est survenue:', error);
  }
}

Design Patterns

# Module Pattern
const counterModule = (function() {
  // Variables privées
  let count = 0;
  
  // Interface publique
  return {
    increment() {
      return ++count;
    },
    decrement() {
      return --count;
    },
    getCount() {
      return count;
    }
  };
})();

console.log(counterModule.getCount());  // 0
counterModule.increment();              // 1
counterModule.increment();              // 2
console.log(counterModule.getCount());  // 2

# Observer Pattern
class Observable {
  constructor() {
    this.observers = [];
  }
  
  subscribe(func) {
    this.observers.push(func);
  }
  
  unsubscribe(func) {
    this.observers = this.observers.filter(observer => observer !== func);
  }
  
  notify(data) {
    this.observers.forEach(observer => observer(data));
  }
}

const news = new Observable();

function emailObserver(data) {
  console.log(`Sending email: ${data}`);
}

function logObserver(data) {
  console.log(`Logging: ${data}`);
}

news.subscribe(emailObserver);
news.subscribe(logObserver);

news.notify("Breaking News!");

Bonnes Pratiques

Code propre

  • Nommage : Utilisez des noms significatifs pour les variables et fonctions
  • Cohérence : Gardez un style de code cohérent (utiliser ESLint et Prettier)
  • KISS : Keep It Simple, Stupid - préférez la clarté à la complexité
  • DRY : Don't Repeat Yourself - évitez la duplication de code
  • Fonctions pures : Préférez les fonctions sans effets de bord
  • Responsabilité unique : Chaque fonction ne devrait faire qu'une seule chose
  • Commentaires : Commentez votre code pour expliquer le "pourquoi", pas le "comment"
  • Tests : Assurez-vous que votre code est testé

# Memoization
// Mémoriser les résultats des appels précédents
function memoize(fn) {
  const cache = {};
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache[key]) {
      return cache[key];
    }
    const result = fn.apply(this, args);
    cache[key] = result;
    return result;
  };
}

const calculCouteux = memoize((x, y) => {
  console.log("Calcul en cours...");
  return x * y;
});

Sécurité

  • XSS : Échappez toujours les données générées par l'utilisateur avant de les insérer dans le DOM
  • CSRF : Utilisez des tokens anti-CSRF pour les requêtes sensibles
  • Content Security Policy : Implémentez une politique de sécurité de contenu
  • HTTPS : Utilisez toujours HTTPS pour protéger les communications
  • Sanitization : Validez et nettoyez toutes les entrées utilisateur
  • Dépendances : Gardez vos dépendances à jour et analysez-les avec des outils comme npm audit
  • JWT : Stockez les tokens JWT dans les cookies HttpOnly plutôt que dans localStorage
  • Eval : Évitez d'utiliser eval() ou Function constructors avec du contenu non fiable
